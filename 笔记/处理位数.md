我来帮你编写一个C语言程序，根据输入的整数求它的高位和低位。这里有几种常见的理解方式：

## 1. 简单版本：获取十进制数的最高位和最低位

```
#include <stdio.h>
#include <stdlib.h>  // 用于abs()函数

int main() {
    int num;
    
    printf("请输入一个整数: ");
    scanf("%d", &num);
    
    // 处理最低位（个位数）
    int low = abs(num) % 10;  // 使用abs()确保负数也能正确获取
    
    // 处理最高位
    int high = abs(num);
    while (high >= 10) {
        high /= 10;
    }
    
    printf("数字: %d\n", num);
    printf("最高位数字: %d\n", high);
    printf("最低位数字: %d\n", low);
    
    return 0;
}
```

## 2. 考虑负数的版本（保留符号）

```
#include <stdio.h>

int main() {
    int num;
    
    printf("请输入一个整数: ");
    scanf("%d", &num);
    
    int sign = 1;
    int temp = num;
    
    if (temp < 0) {
        sign = -1;
        temp = -temp;  // 转为正数处理
    }
    
    // 最低位
    int low = temp % 10;
    
    // 最高位
    int high = temp;
    while (high >= 10) {
        high /= 10;
    }
    
    printf("\n结果:\n");
    printf("原始数字: %d\n", num);
    printf("最高位数字: %d\n", sign * high);
    printf("最低位数字: %d\n", low);
    
    return 0;
}
```

## 3. 处理多字节整数的高位和低位（二进制表示）

如果你指的是从二进制角度获取高位和低位字节：

```
#include <stdio.h>

int main() {
    unsigned int num;
    unsigned int high_byte, low_byte;
    
    printf("请输入一个整数: ");
    scanf("%u", &num);
    
    // 获取低8位（低字节）
    low_byte = num & 0xFF;
    
    // 获取高8位（高字节）
    // 注意：这里假设是32位整数，高位字节是最高8位
    high_byte = (num >> 24) & 0xFF;
    
    printf("\n结果:\n");
    printf("输入的数字: %u (十六进制: 0x%08X)\n", num, num);
    printf("高字节: 0x%02X (%u)\n", high_byte, high_byte);
    printf("低字节: 0x%02X (%u)\n", low_byte, low_byte);
    
    return 0;
}
```

## 4. 16位整数的高低位（更常见的需求）

```
#include <stdio.h>

int main() {
    unsigned short num;  // 16位整数
    unsigned char high, low;
    
    printf("请输入一个0-65535之间的整数: ");
    scanf("%hu", &num);
    
    // 低8位
    low = num & 0xFF;
    
    // 高8位
    high = (num >> 8) & 0xFF;
    
    printf("\n结果:\n");
    printf("输入的数字: %hu\n", num);
    printf("高位字节: %u (0x%02X)\n", high, high);
    printf("低位字节: %u (0x%02X)\n", low, low);
    printf("组合后的值: %hu\n", (high << 8) | low);
    
    return 0;
}
```

## 5. 通用函数版本

```
#include <stdio.h>

// 获取十进制数的最高位
int getHighestDigit(int n) {
    if (n < 0) n = -n;  // 处理负数
    while (n >= 10) {
        n /= 10;
    }
    return n;
}

// 获取十进制数的最低位
int getLowestDigit(int n) {
    if (n < 0) n = -n;  // 处理负数
    return n % 10;
}

// 获取整数的高位字节（16位整数）
unsigned char getHighByte(unsigned short n) {
    return (n >> 8) & 0xFF;
}

// 获取整数的低位字节（16位整数）
unsigned char getLowByte(unsigned short n) {
    return n & 0xFF;
}

int main() {
    int choice, num;
    
    printf("请选择模式:\n");
    printf("1. 十进制数字的高位和低位\n");
    printf("2. 二进制表示的高位和低位字节\n");
    printf("请选择: ");
    scanf("%d", &choice);
    
    if (choice == 1) {
        printf("请输入一个整数: ");
        scanf("%d", &num);
        
        printf("\n十进制分析:\n");
        printf("数字: %d\n", num);
        printf("最高位数字: %d\n", getHighestDigit(num));
        printf("最低位数字: %d\n", getLowestDigit(num));
    } 
    else if (choice == 2) {
        unsigned short num2;
        printf("请输入一个0-65535的整数: ");
        scanf("%hu", &num2);
        
        printf("\n二进制分析:\n");
        printf("数字: %hu (0x%04X)\n", num2, num2);
        printf("高位字节: 0x%02X (%u)\n", getHighByte(num2), getHighByte(num2));
        printf("低位字节: 0x%02X (%u)\n", getLowByte(num2), getLowByte(num2));
    }
    
    return 0;
}
```

## 选择建议

1. 如果你说的是**十进制数字**的最高位和最低位，用**第一个或第二个**程序
    
2. 如果你说的是**二进制字节**的高位和低位，用**第三个或第四个**程序
    
3. 如果你不确定，可以用**第五个**程序，它提供了两种模式
    

请根据你的具体需求选择合适的版本！
